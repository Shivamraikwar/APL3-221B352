class One {
    One(int x) {
        System.out.println("Class One constructor with parameter: " + x);
    }
}

class Two extends One {
    Two(int x) {
        super(x);  
        System.out.println("Class Two constructor with parameter: " + x);
    }
}

public class Main {
    public static void main(String[] args) {
        Two obj = new Two(10);
    }
}
